import { Prop, Schema, SchemaFactory } from '@nestjs/mongoose';
import { Document } from 'mongoose';

export type SatelliteDocument = Satellite & Document;

interface RemoteModule {
    moduleId: string;
    path: string;
}
class PublicInformation {
    @Prop()
    publicData: boolean;

    @Prop()
    connected: boolean;

    @Prop()
    domain: string;

    @Prop()
    common_project_id: string;

    @Prop({ type: Object })
    remoteModules: RemoteModule[];

    @Prop()
    absoluteUrl: string;

    @Prop()
    localip: string;

    @Prop()
    port: string;

    @Prop()
    version?: string;

    @Prop()
    newAbsoluteUrl?: string;

    @Prop()
    expiry_time_ms: number;

    @Prop()
    ownerProvidedUrl?: string;

    @Prop()
    maxSimultaneousTasks?: number;

    @Prop()
    publicIp?: string;
}

class AbsoluteUrlSettings {
    @Prop()
    origin: string; // automatic, fixed, owner

    @Prop()
    inactiveTimerMS: number; // Time in MS before Satellite gets unassigned

    @Prop()
    providedCerts?: boolean; // Does the owner provide their own certs

    @Prop()
    key?: string;

    @Prop()
    cert?: string;

    @Prop()
    fixedIp?: boolean;
}


@Schema({ collection: 'satellites' })
export class Satellite {
    @Prop()
    _id: string;

    @Prop()
    labId: string;

    @Prop()
    type: string;

    @Prop({ type: Object })
    public: PublicInformation;

    @Prop()
    token: string;

    @Prop({ type: Object })
    auth: any;

    @Prop({ type: Object })
    sync: any;

    @Prop({ type: Object })
    absoluteUrlSettings: AbsoluteUrlSettings;

    @Prop({ type: Object })
    satMachines: any;

    @Prop({ type: Object })
    satTasks: any;

    @Prop({ type: Object })
    apiMachines: any;

    @Prop()
    scidapHome: string;

    @Prop({ type: Array })
    configOverwrites: any[];
}

export const SatelliteSchema = SchemaFactory.createForClass(Satellite);
