import { Injectable } from '@nestjs/common';
import { InjectModel } from '@nestjs/mongoose';
import { Model } from 'mongoose';

import { Lab, LabDocument } from './schemas/lab.schema';

@Injectable()
export class LabService {
    constructor(
        @InjectModel(Lab.name)
        private labModel: Model<LabDocument>,
    ) {}

    /**
     * Get a Lab from the database by ID
     * @param _id mongo id of the Lab
     */
    async getById(_id: string): Promise<Lab> {
        return this.labModel.findOne({ _id }).exec();
    }

    /**
     * Update one Lab by ID
     * @param _id id of the lab
     * @param updatedObject object to update
     * @returns error or amount upserted (which is one)
     */
     public async updateOne(_id: string, updatedObject: any) {
        try {
            const lab = await this.labModel.updateOne({ _id }, { $set: updatedObject });
            return { upserted: lab.nModified };
        } catch (err) {
            return { err };
        }
    }
}
