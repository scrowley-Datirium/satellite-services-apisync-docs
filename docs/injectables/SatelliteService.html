import { Injectable } from '@nestjs/common';
import { InjectModel } from '@nestjs/mongoose';
import { Model } from 'mongoose';
import { LocalFiles } from '../interfaces';

import { Satellite, SatelliteDocument } from './schemas/satellite.schema';
import Log from '../lib/logger';

@Injectable()
export class SatelliteService {
    constructor(
        @InjectModel(Satellite.name)
        public satelliteModel: Model<SatelliteDocument>,
    ) {}


    public async updateOne(_id: string, updatedObject: any) {
        try {
            const sat = await this.satelliteModel.updateOne({ _id }, { $set: updatedObject });
            return { upserted: sat.nModified };
        } catch (err) {
            return { err };
        }
    }

    /**
     * Get a satellite by its token
     * @param token unique JWT for satellite
     */
    async getByToken(token: string): Promise<Satellite> {
        try {
            const satellite = await this.satelliteModel.findOne({ token });
            if (!satellite.absoluteUrlSettings) {
                const settings = {
                    origin: 'automatic',
                    inactiveTimerMS: 14 * 24 * 60 * 60 * 1000,
                };
                return await this.satelliteModel.findByIdAndUpdate(satellite._id, { $set: { absoluteUrlSettings: settings } }, { new: true, useFindAndModify: false });
            }
            return satellite;
        } catch (err) {
            throw err;
        }
    }

    async getById(_id: string) {
        return this.satelliteModel.findOne({ _id });
    }

    /**
     * Check if the satellite exists and can establish a link
     *  This is where we would check if satellite can connect
     *  THis is used by the gRPC Guard
     * @param token unique JWT for satellite
     */
    async canLink(token: string): Promise<boolean> {
        try {
            const satellite = await this.satelliteModel.findOne({ token });
            if (!satellite) throw Error(`Satellite not found token: ${token}`);
            // TODO: Check all the other things you want to check
            return true;
        } catch (err) {
            throw err;
        }
    }

    /**
     * Update the version of satellite in DB to current
     * @param _id of satellite
     * @param version of satellite
     */
    async updateVersion(_id: string, version: string) {
        try {
            return this.satelliteModel.findOneAndUpdate({ _id }, { $set: { 'public.version': version } }, { useFindAndModify: false });
        } catch (err) {
            throw err;
        }
    }

    /**
     * Update the version of satellite in DB to current
     * @param _id of satellite
     * @param version of satellite
     * @deprecated This is old, we should be doing all of this from central server not other way
     */
    async updateLocalFiles(_id: string, localFiles: LocalFiles) {
        try {
            const sample = await this.satelliteModel.findById(_id);
            const remotes = sample.public.remoteModules;
            let updated = false;
            const updatedRemotes = remotes.map((remote) => {
                const updatedRemote = { ...remote };
                if (updatedRemote.moduleId === 'localfiles') {
                    updatedRemote.path = localFiles.collection[0].path;
                    updated = true;
                }
                return updatedRemote;
            });
            if (!updated) updatedRemotes.push({ moduleId: 'localfiles', path: localFiles.collection[0].path });

            return this.satelliteModel.findOneAndUpdate(
                { _id },
                { $set: { 'public.showLocalFiles': localFiles.show, 'public.remoteModules': updatedRemotes } },
                { useFindAndModify: false },
            );
        } catch (err) {
            throw err;
        }
    }

    /**
     * Updates the expiry_time_ms of a satellite to a time two weeks from now
     * @param _id satellite Id
     * @param disconnect boolean for if the satellite is disconnecting
     * @param timeToExpire number time in MS till expiration
     */
    async updateTime(_id: string, disconnect: boolean, timeToExpire?: number) {
        try {
            let expiryDate = Infinity;
            if (disconnect) {
                if (!!timeToExpire) {
                    expiryDate = new Date().getTime() + timeToExpire;
                } else {
                    const twoWeeksMS = 14 * 24 * 60 * 60 * 1000;
                    expiryDate = new Date().getTime() + twoWeeksMS;
                }
            }
            const doc = await this.satelliteModel.findOneAndUpdate({ _id }, { $set: { 'public.expiry_time_ms': expiryDate } }, { useFindAndModify: false });
            if (!doc) {
                throw new Error('Satellite Not Found');
            }
            return new Date(expiryDate);
        } catch (err) {
            throw err;
        }
    }

    async getUnusedSatellites(): Promise<Satellite[]> {
        try {
            const timeMS = new Date().getTime();
            return await this.satelliteModel.find({ 'public.expiry_time_ms': { $lte: timeMS } }).exec();
        } catch (err) {
            throw err;
        }
    }

    /**
     * 
     * @param satId Id of Satellite to Update
     * @param ips Local IP Address of Satellite (used to update Digital Ocean Record)
     * @returns Updated Satellite
     */
    async updateIPs(satId: string, ips: string) {
        try {
            return this.satelliteModel.findOneAndUpdate({ _id: satId }, { $set: { 'public.localip': ips } }, { useFindAndModify: false });
        } catch (err) {
            throw err;
        }
    }

    /**
     * Updates the Connected Status of a Satellite and also set's the public IP address where it is conencting from
     * @param satId Id of the Satellite
     * @param connected Whether the Satellite is connected
     * @param publicIP Public Facing IP Address of Satellite
     * @returns 
     */
    async updateConnectedStatus(satId: string, connected: boolean, publicIP?: string): Promise<boolean> {
        try {
            if (publicIP) {
                await this.satelliteModel.findByIdAndUpdate(satId, { $set: { 'public.connected': connected, 'public.publicIp': publicIP } }, { useFindAndModify: false });
            }
            await this.satelliteModel.findByIdAndUpdate(satId, { $set: { 'public.connected': connected } }, { useFindAndModify: false });
            return true;
        } catch (err) {
            Log.error(`Failed to update connection status for ${satId}: ${err}`);
            return false;
        }
    }

    async updateNewAbsoluteUrl(satId: string, newUrl: string) {
        try {
            if (!newUrl.includes('https://')) {
                newUrl = 'https://' + newUrl;
            }
            return this.satelliteModel.findByIdAndUpdate(satId, { $set: { 'public.newAbsoluteUrl': newUrl } }, { useFindAndModify: false });
        } catch (err) {
            throw err;
        }
    }
}
