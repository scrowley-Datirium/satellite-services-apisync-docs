import { forwardRef, Inject, Injectable } from '@nestjs/common';
import { InjectModel } from '@nestjs/mongoose';
import { Model } from 'mongoose';

// import { updatedInfoTasksQueue } from '../interfaces';
import { TasksQueueDocument, TasksQueue } from './schemas/task.schema';
import { SamplesService } from './samples.service';
import Log from '../lib/logger';
import { ServiceTag, ServiceTagDocument } from './schemas/serviceTag.schema';

@Injectable()
export class TasksQueueService {
    constructor(
        @InjectModel(TasksQueue.name)
        private tasksQueueModel: Model<TasksQueueDocument>,
        @Inject(forwardRef(() => SamplesService))
        private sampleService: SamplesService,
        @InjectModel(ServiceTag.name)
        private serviceTagModel: Model<ServiceTagDocument>,
    ) {}

    /**
     * Find all of the tasks for a given satellite
     * @param satellite_id mongo id of the satellite
     */
    async getBySatId(satellite_id: string): Promise<{ queue: TasksQueueDocument[]; priorityQueue: TasksQueueDocument[] }> {
        const priorityTag = await this.serviceTagModel.findOne({ priority: true }).exec();
        const tasks = await this.tasksQueueModel.find({ satellite_id }).exec();
        if (!priorityTag) return { queue: tasks, priorityQueue: [] };
        return tasks.reduce(
            (final, task) => {
                const service_tags = task.payload.get('service_tags');
                if (!service_tags) final.queue.push(task);
                else if (task.payload.get('service_tags').includes(priorityTag._id)) final.priorityQueue.push(task);
                else final.queue.push(task);
                return final;
            },
            { queue: [], priorityQueue: [] },
        );
    }

    /**
     * Get task by id, must include matching satellite id
     * @param task_id mongo id of task
     * @param satellite_id mongo id of satellite
     */
    async getById(task_id: string, satellite_id: string): Promise<TasksQueue> {
        try {
            const task = await this.tasksQueueModel.findById(task_id).exec();
            if (!task) throw Error(`Task ${task_id} not found`);
            if (task.satellite_id === satellite_id) return task;
            throw Error(`${satellite_id} attempted to access wrong task`);
        } catch (err) {
            throw err;
        }
    }

    /**
     * Delete a task by id and satellite id
     * @param _id mongo id of task
     * @param satellite_id mongo id of satellite
     */
    async delete(_id: string, satellite_id: string): Promise<void> {
        try {
            const task = await this.tasksQueueModel.findById(_id).exec();
            if (!task) throw Error(`Task ${_id} not found`);
            if (task.satellite_id !== satellite_id) throw Error('Attempted to access wrong task');
            const tasksBySat = await this.tasksQueueModel.find({ satellite_id: task.satellite_id });
            tasksBySat.forEach(async (t) => {
                if (t._id === task._id) return;
                if (t.date.created < task.date.created) {
                    console.log(`Task ${t.id} Earlier than ${task.id}`);
                    return;
                }
                try {
                    const sample = await this.sampleService.getById(t.payload.get('sample_id'));
                    const visualPlugin = { ...sample.preview.visualPlugins[0] };
                    const num = visualPlugin.progress.number - 1;
                    if (!num) return;
                    visualPlugin.progress = {
                        ...visualPlugin.progress,
                        title: num === 1 ? 'Starting to Process' : `${num}${this.nth(num)} in Queue`,
                        number: num,
                    };
                    Log.debug(`Updated queue location for ${sample._id} to ${num}`);
                    this.sampleService.updatePreview(sample._id, visualPlugin);
                } catch (err) {
                    Log.error(`Error updating Queue Location for ${t.payload.get('sample_id')}: ${err}`);
                }
            });
            await this.tasksQueueModel.deleteOne({ _id });
        } catch (err) {
            throw err;
        }
    }

    /**
     * Update the progress of a given task
     * @param task_id mongo id of task
     * @param updatedInfo task info to update
     */
    // async update(task_id: string, updatedInfo: updatedInfoTasksQueue): Promise<TasksQueue> {
    async update(task_id: string, updatedInfo: any): Promise<TasksQueue> {
        const newUpdatedInfo = { ...updatedInfo };
        const task = await this.tasksQueueModel.findById(task_id);
        const updatedPayload = new Map();
        task.payload.forEach((value, key) => {
            updatedPayload.set(key, value);
        });
        if (newUpdatedInfo.payload) {
            Object.keys(newUpdatedInfo.payload).forEach((key) => {
                updatedPayload.set(key, newUpdatedInfo.payload[key]);
            });
            newUpdatedInfo.payload = updatedPayload;
        }
        return await this.tasksQueueModel.findByIdAndUpdate(task_id, { ...newUpdatedInfo }, { useFindAndModify: false }).exec();
    }

    /**
     * Update Download Progress by Task ID and File ID
     * @param taskId mongo id of task
     * @param fileId id to find file to update
     * @param status status of download
     * @param progress 0 to 100 represents download progress
     * @param satId mongo id of satellite
     */
    async updateDownloadProgress(taskId: string, fileId: string, status: string, progress: number, satId: string) {
        try {
            const task = await this.getById(taskId, satId);
            const downloads = task.payload.get('downloads');
            const download = { ...downloads[fileId] };
            if (status === 'finished') {
                download.downloaded = true;
            } else {
                download.progress = progress;
            }
            const toUpdate = `payload.downloads.${fileId}`;
            await this.tasksQueueModel.findByIdAndUpdate(taskId, { $set: { [toUpdate]: download } }, { useFindAndModify: false });

            const preview = {
                state: status,
                dagId: '',
                runId: task.payload.get('sample_id'),
                progress: progress,
            };
            await this.sampleService.updateProgress(preview);
        } catch (err) {
            throw err;
        }
    }

    private nth(n: number): string {
        return [, 'st', 'nd', 'rd'][(n / 10) % 10 ^ 1 && n % 10] || 'th';
    }

    /**
     * Finds a Task Queue Object by Sample Id and Satellite Id
     * @param sampleId Id of sample to find task by
     * @param satelliteId Satellite Id that task belongs to
     * @returns A Task Queue Object
     */
    async findTaskBySampleId(sampleId: string, satelliteId: string): Promise<TasksQueue> {
        try {
            const task = await this.tasksQueueModel.findOne({ satellite_id: satelliteId, 'payload.sample_id': sampleId }).exec();
            if (!task) throw new Error(`No Task with sample: ${sampleId} for satellite: ${satelliteId}`);
            return task;
        } catch (err) {
            throw err;
        }
    }
}
