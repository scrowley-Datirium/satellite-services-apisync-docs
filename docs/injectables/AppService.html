import { forwardRef, Inject, Injectable } from '@nestjs/common';
import { Cron, CronExpression } from '@nestjs/schedule';
import { Subject } from 'rxjs';
const fs = require('fs');
const path = require('path');

import { APISyncMessage, CreateAPIMachine, SatelliteConnection, SendEvent, TriggerContext } from './interfaces';
import { StateMachinesService } from './stateMachines/stateMachines.service';
import Log from './lib/logger';
import { SatelliteService } from './mongoConnection/satellite.service';
import { TasksQueueService } from './mongoConnection/tasksQueue.service';
import { SamplesService } from './mongoConnection/samples.service';
import { TasksQueueDocument } from './mongoConnection/schemas/task.schema';

@Injectable()
export class AppService {
    public sendToSat$ = new Subject<APISyncMessage>();
    public sendToAPI = new Subject<SendEvent>();
    public triggerContext = new Subject<TriggerContext>();
    public createFSM = new Subject<CreateAPIMachine>();
    public removeFSM = new Subject<{ satelliteId: string; machine: string, callback?: (err: string) => void }>()
    public satConnections: { [key: string]: SatelliteConnection } = {};

    constructor(
        @Inject(forwardRef(() => StateMachinesService))
        private _stateMachineService: StateMachinesService,
        private _satelliteService: SatelliteService,
        private _tasksQueueService: TasksQueueService,
        private _samplesService: SamplesService,
        ) {
        this._stateMachineService.dbStream$.subscribe((msg) => {
            console.log('Write to db', msg);
        });
    }

    /** Seperate Config File Test */
    public async seperateConfigFiles(): Promise<any[]> {
        const loc = path.join(__dirname, process.env.TEST_FILE);
        const machines = [];
        return new Promise((resolve, reject) => {
            fs.readFile(loc, 'utf-8', (err, data) => {
                if (err) {
                    console.error(err);
                    reject(err);
                }
                // console.log(data);
                const configs = JSON.parse(data);
                // console.log(configs)
                Object.keys(configs).forEach((key) => {
                    machines.push(configs[key]);
                });
                resolve(machines);
            });
        });
    }

    /**
     * Heartbeat for API-Sync that checks for new tasks in queue for connected satellites
     *  If a satellite has room for a task it checks the queue,
     *  If not it checks if the current task has been updated
     */
    @Cron(CronExpression.EVERY_5_SECONDS)
    async asyncheartbeat() {
        try {
            const satKeys = Object.keys(this.satConnections);
            for (const key of satKeys) {
                const satConnection = this.satConnections[key];
                const satId = satConnection.satInformation._id;
                Log.debug(`Checking the Queue for satellite ${satId}`);
                // Get all the regular queue and priority queue tasks for the satellite
                const { queue, priorityQueue } = await this._tasksQueueService.getBySatId(satId);
                // Get all the FSMs that run tasks on the current satellite
                const taskRunners = Object.keys(satConnection.stateMachines)
                    .filter((k) => k.includes('task_queue'))
                    .map((k) => satConnection.stateMachines[k].fsm);
                // Get all the waiting machine on the satellite
                const waitingMachines = taskRunners.filter((fsm) => fsm.state.value === 'waiting');
                Log.debug(`${waitingMachines.length} machines currently waiting out of ${taskRunners.length}`);
                if (waitingMachines.length < 1) continue;
                // Get all of the tasks that are currently running
                const currentTasks = taskRunners
                    .filter((fsm) => fsm.state.value !== 'waiting')
                    .map((fsm) => fsm.state.context.taskId);

                let taskCnt = 0, priorityCnt = 0;
                const sortedTasks = queue
                    .filter((task) => {
                        const res = !currentTasks.includes(task._id);
                        if (!res) taskCnt++;
                        return res; 
                    })
                    .filter((task) => task.action !== 'subscription_cancelled')
                    .sort((a, b) => a.date.modified - b.date.modified);
                const sortedPriorityTasks = priorityQueue
                    .filter((task) => {
                        const res = !currentTasks.includes(task._id);
                        if (!res) priorityCnt++;
                        return res;
                    })
                    .filter((task) => task.action !== 'subscription_cancelled')
                    .sort((a, b) => a.date.modified - b.date.modified);

                for (const machine of waitingMachines) {
                    let task: TasksQueueDocument;
                    const half = Math.floor(satConnection.maxTasks / 2);
                    // If half the tasks are standard AND there are priority tasks add priority
                    while ((sortedTasks.length > 0 || sortedPriorityTasks.length > 0) || !task) {
                        let checking: TasksQueueDocument;
                        let priority = false;
                        if ((taskCnt >= half || sortedTasks.length === 0) && sortedPriorityTasks.length > 0) {
                            checking = sortedPriorityTasks.shift();
                            priority = true;
                        } else {
                            checking = sortedTasks.shift();
                            priority = false;
                        }
                        if (!checking) break;
                        const sampleId = checking.payload.get('sample_id');
                        // Check to make sure the upstreams are ready
                        const upstreamReady = await this._samplesService.checkUpstreamOutputs(sampleId);
                        // If they are save this task as the one to add to the task Runner and increment regular task cnt
                        if (upstreamReady) {
                            task = checking;
                            if (priority) priorityCnt++;
                            else taskCnt++;
                            break;
                        }
                    }
                    if (task) {
                        Log.info(`Sending ${task._id} to ${machine.state.context.id}`);
                        machine.state.context.startProcess(task._id);
                    }
                }
            }
        } catch (err) {
            Log.error(`Error with heartbeat: ${err}`);
        }
    }
    
    /**
     * 
     * @param states configuration for state machine
     * @param options actions and guards for state machine
     * @param transitions actions to perfom on transition states
     * @param satelliteId 
     * @returns @param {ProcessedStateMachine} fsm Interpreted Finite State Machine and a specific Logger for this State Machine 
     */
   public async setupStateMachine(states: any, options: any, transitions: any, satelliteId: string) {
       const fsm = await this._stateMachineService.createStateMachine(states, options, transitions, satelliteId);
       return fsm;
   }
