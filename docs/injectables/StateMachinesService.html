import { forwardRef, Inject, Injectable } from '@nestjs/common';
import { HttpService } from '@nestjs/axios';
import { ConfigService } from '@nestjs/config';
import { Subject } from 'rxjs';
import { CWLService } from '../mongoConnection/cwl.service';
import { createMachine, interpret, Interpreter } from 'xstate';
const path = require('path');
import fs = require('fs');

import { ProcessedStateMachine, SendEvent, SatelliteTask, StoreKey, APISyncMessage, ServiceCommand } from '../interfaces';
import { stateLogger } from '../lib/logger';
import { SamplesService } from '../mongoConnection/samples.service';
import { SatelliteService } from '../mongoConnection/satellite.service';
import { SubDomainService } from '../mongoConnection/subDomain.service';
import { TasksQueueService } from '../mongoConnection/tasksQueue.service';
import { LabService } from '../mongoConnection/labs.service';
import { AppService } from '../app.service';

@Injectable()
export class StateMachinesService {
    constructor (
        private _cwlService: CWLService,
        private _sampleService: SamplesService,
        private _satelliteService: SatelliteService,
        private _subDomainService: SubDomainService,
        private _tasksQueueService: TasksQueueService,
        private _labService: LabService,
        private _httpService: HttpService,
        private _configService: ConfigService,
        @Inject(forwardRef(() => AppService))
        private _appService: AppService,
    ) {}
    // Currently does nothing real
    public dbStream$ = new Subject<any>();

    // Subject that sends tasks to the appropriate Satellite
    public satStream$ = new Subject<APISyncMessage>();

    // Subject that sends messages to other State Machines relevant to the same satellite
    public sendEvent$ = new Subject<SendEvent>();

    // Subject that stores a key value pair on a state machien
    public storeValue$ = new Subject<StoreKey>();

    private _path = path;
    private _fs = fs;
    private _fsPromises = fs.promises;

    /**
     * Create a State Machine and setup transitions for it
     * @param config Contains config setup for State Machine
     * @param options Contains Actions and Guards
     * @param transitions Actions to perform on Transition States
     * @param satelliteId Id of the relevant Satellite
     * @returns Interpreted Finite State Machine and a specific Logger for this State Machine
     */
    public async createStateMachine(config: any, options: any, transitions: any, satelliteId: string): Promise<ProcessedStateMachine> {
        const updatedOptions = { ...options };
        const updatedConfig = { ...config };
        const logger = await stateLogger(config.id, satelliteId);
        
        const AsyncFunction = Object.getPrototypeOf(async function(){}).constructor

        // Update the config for any keys that are marked as functions
        const updatedContext = {};
        if (config.context) {
            Object.keys(config.context).forEach((k) => {
                const value = config.context[k];
                if (typeof value === 'string' && value.includes('func::')) {
                    updatedContext[k] = (args) => {
                        try {
                            const script = value.replace('func::', '');
                            // Don't require the target if sending to itself
                            const send = (type: string, target?: string | number, delay?: number) => {
                                let to: string;
                                if (!target || typeof target === 'number') {
                                    to = config.id;
                                    if (typeof target === 'number') {
                                        delay = target;
                                    }
                                } else {
                                    to = target;
                                }
                                this.sendEvent$.next({ to, type, delay, satelliteId });
                            };
                            const satTasks = (tasks: SatelliteTask[]) => this.satStream$.next({ satelliteid: satelliteId, tasks });
                            const satCommands = (commands: ServiceCommand[]) => this.satStream$.next({ satelliteid: satelliteId, commands });
                            // Don't require the target if storing on itself
                            const store = (key: string, value: any, target?: string, func?: any) => {
                                let to = target || config.id;
                                this.storeValue$.next({ satelliteId, to, key, value })
                            };
                            const context = (key: string) => this._appService.satConnections[satelliteId].stateMachines[config.id].fsm.state.context[key];
                            new AsyncFunction('logger', 'satelliteId', 'args', 'send', 'satTasks', 'satCommands', 'store', 'context', script).call(this, logger, satelliteId, args, send, satTasks, satCommands, store, context);
                        } catch (err) {
                            logger.error(`Failed to run context ${k}: ${err}`);
                        }
                    }
                } else {
                    updatedContext[k] = value;
                }

            });
        }

        // Update the actions to allow the functions to run correctly
        const updatedActions = {};
        if (updatedOptions.actions) {
            const actions = updatedOptions.actions;
            Object.keys(actions).forEach(
                (k) => updatedActions[k] = (ctx) => {
                        try {
                            // Don't require the target if sending to itself
                            const send = (type: string, target?: string | number, delay?: number) => {
                                let to: string;
                                if (!target || typeof target === 'number') {
                                    to = config.id;
                                    if (typeof target === 'number') {
                                        delay = target;
                                    }
                                } else {
                                    to = target;
                                }
                                this.sendEvent$.next({ to, type, delay, satelliteId });
                            };
                            new AsyncFunction('ctx', 'logger', 'satelliteId', 'send', actions[k]).call(this, ctx, logger, satelliteId, send);
                        } catch (err) {
                            logger.error(`Failed to run action ${k}: ${err}`);
                        }
                    }
            );
        }
        
        // Update teh guards to allow the functions to run correctly
        const updatedGuards = {};
        if (updatedOptions.guards) {
            const guards = updatedOptions.guards;
            Object.keys(guards).forEach(
                (k) =>
                    (updatedGuards[k] = (ctx) => {
                        try {
                            return new AsyncFunction('ctx', 'logger', 'satelliteId', guards[k]).call(this, ctx, logger, satelliteId);
                        } catch (err) {
                            logger.error(`Failed to run guard ${k}: ${err}`);
                        }
                    }),
            );
        }

        updatedOptions.actions = updatedActions;
        updatedOptions.guards = updatedGuards;
        
        updatedConfig.context = updatedContext;

        const machine = createMachine(updatedConfig, updatedOptions);
        const interpreted = interpret(machine);

        // Correctly Parse the functions for Transitions
        const updatedTransitions = {};
        if (transitions) {
            Object.keys(transitions).forEach(
                (k) =>
                    (updatedTransitions[k] = (state) => {
                        try {
                            new AsyncFunction('state', 'logger', 'satelliteId', transitions[k]).call(this, state, logger, satelliteId);
                        } catch (err) {
                            logger.error(`Failed to run transition ${k}: ${err}`);
                        }
                    }),
            );
        }

        interpreted.start();

        // Setup the Transitions for the StateMachine
        interpreted.onTransition((state) => {
            logger.info(`Current State is ${state.value}`);
            if (typeof state.value !== 'string') {
                logger.error(`Invalid state: ${state.value} - type ${typeof state.value}`);
                return;
            }
            if (updatedTransitions[state.value]) {
                try {
                    updatedTransitions[state.value](state);
                } catch (err) {
                    logger.error(`Error with transition to ${state.value}: ${err}`);
                }
            } else {
                logger.debug(`No custom action for ${state.value}`);
            }
        });

        return { interpreted, logger };
    }
}
