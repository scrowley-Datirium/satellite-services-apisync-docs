    constructor(
        private _appService: AppService,
        private _stateMachineService: StateMachinesService,
        private _satelliteService: SatelliteService,
        private _subDomainService: SubDomainService,
        private _httpService: HttpService,

        // Create a state machine on API-Sync relevant to a Satellite
        this._appService.createFSM.subscribe(async ({ config, options, transitions, satelliteId, callback }) => {
            if (!satelliteId) satelliteId = 'test_satellite';
            const { interpreted, logger } = await this._appService.setupStateMachine(config, options, transitions, satelliteId);
            this._appService.satConnections[satelliteId].stateMachines[config.id] = { fsm: interpreted, logger };
            if (callback) {
                callback();
            }
        });

        this._appService.removeFSM.subscribe(({ machine, satelliteId, callback }) => {
            if (!satelliteId) satelliteId = 'test_satellite';
            try {
                delete this._appService.satConnections[satelliteId].stateMachines[machine];
                if (callback) {
                    callback(null);
                }
            } catch (err) {
                const message = `Error removing ${machine} from ${satelliteId}: ${err}`;
                Log.error(message);
                if (callback) {
                    callback(message);
                }
            }
        });

        merge(this._appService.sendToSat$, this._stateMachineService.satStream$).pipe(
            tap((m) => Log.info(`Sending ${!!m.trees?.length ? 'Trees' : ''} ${!!m.tasks?.length ? 'Tasks' : ''} to ${m.satelliteid}`)),
        ).subscribe((message) => {
            let { satelliteid, trees, tasks, commands } = message;
            if (!satelliteid) satelliteid = 'test_satellite';
            try {
                this._appService.satConnections[satelliteid].stream.write({ tasks, trees, commands });
            } catch (err) {
                Log.error(`Failed to send message to ${satelliteid}: ${err}`);
            }
        });

        merge(this._stateMachineService.sendEvent$, this._appService.sendToAPI)
            .pipe(
                tap(({ to, type, delay, satelliteId }) => Log.info(`${satelliteId} is attempting to send ${type} to ${to} ${!delay ? '' : `with ${delay}ms delay`}`)),
                filter(({ to, satelliteId }) => Object.keys(this._appService.satConnections[satelliteId].stateMachines).includes(to)),
            )
            .subscribe(async ({ to, type, delay, satelliteId }) => {
                if (delay) {
                    await new Promise((r) => setTimeout(() => r(true), delay));
                    Log.info(`${satelliteId} is sending ${type} to ${to}`);
                    this._appService.satConnections[satelliteId].stateMachines[to].fsm.send({ type });
                } else {
                    Log.info(`${satelliteId} is sending ${type} to ${to}`);
                    this._appService.satConnections[satelliteId].stateMachines[to].fsm.send({ type });
                }
            });
        
        this._appService.triggerContext
            .pipe(
                tap(({ to, satelliteId, method }) => Log.info(`${satelliteId} is attempting to trigger ${method} on ${to}`)),
            )
            .subscribe(async ({ to, satelliteId, method, callback, args }) => {
                const err: string = await new Promise((resolve) => {
                    try {
                        const func = this._appService.satConnections[satelliteId].stateMachines[to].fsm.state.context[method];
                        if (!func) throw new Error(`${method} not found`);
                        func(args);
                        resolve(null);
                    } catch (err) {
                        const msg = `Error running ${method} on ${satelliteId}-${to}: ${err}`;
                        resolve(msg);
                    }
                });
                if (callback) {
                    callback(err);
                }
            });

        this._stateMachineService.storeValue$
            .pipe(
                tap(({ to, key, satelliteId }) => Log.info(`${satelliteId} is attempting to store ${key} on ${to}`)),
                filter(({ to, satelliteId }) => Object.keys(this._appService.satConnections[satelliteId].stateMachines).includes(to)),
            )
            .subscribe(async ({ to, key, value, satelliteId }) => {
                Log.info(`${satelliteId} is storing ${key} on ${to}`);
                this._appService.satConnections[satelliteId].stateMachines[to].fsm.state.context[key] = value;
            });

    }

                Log.debug(`${satId} expires on ${expiryDate.toUTCString()}`);
            } catch (err) {
                Log.error(`Failed to update Expiry Time: ${err.message}`);
            } finally {
                delete this._appService.satConnections[satId];
            }
        })
    }

    /**
     * This function runs a timer that if it isn't called in that time it breaks the connection and restarts it
     * @param satConnection Connection to the given satellite
     */
    runKeepAliveTimer(satConnection: SatelliteConnection) {
        clearTimeout(satConnection.keepAliveTimer);
        satConnection.keepAliveTimer = setTimeout(() => {
            Log.info(`Did not recieve Keep Alive Ping from ${satConnection.satInformation._id}`);
            clearInterval(satConnection.keepAlivePingTimer);
            satConnection.stream.end();
        }, pingKillTimer);
    }
    

    @UseFilters(ExceptionFilter) // Throws exceptions when auth fails
    @GrpcStreamCall('APISyncService', 'FileStream')
    async fileUpload(requestStream, callback) {
        const tempName = crypto.randomBytes(20).toString('hex');
        let fileName;
        const tempPath = path.join(__dirname, '/tmp/');
        const finalPath = path.join(__dirname, '/final/');
        await promises.mkdir(tempPath, { recursive: true });
        await promises.mkdir(finalPath, { recursive: true });
        await promises.open(tempPath + tempName, 'w');
        const writeStream = createWriteStream(tempPath+ tempName);
        requestStream.on('data', (msg) => {
            writeStream.write(msg.file);
            if (msg.file_name && !fileName) {
                fileName = msg.file_name
            }
        });

        requestStream.on('end', (msg) => {
            console.log('File Finished');
            callback(null, { type: 'SERVER', status: 'SUCCESS' });
            rename(tempPath + tempName, finalPath + fileName, () => {
                console.log('Success');
            });
        });
    }

    /**
     * Processes a single message from Satellite by sending it to a relevant State Machine
     *  on API-Sync
     * @param target Name of the service to send to
     * @param action Name of the action to run
     * @param message Arguments to send to the function
     * @param satConnection Information about the Satellite Connection
     */
    processMessagesFromMainService(target: string, action: string, message: string, satConnection: SatelliteConnection) {
        if (!satConnection.stateMachines[target]) {
            Log.error(`No State Machine named ${target}, failed to send: ${action} - ${message}`);
            return;
        }
        try {
            // Locate the action in the machine
            const upAction = action.split('.').reduce((acum, curr) => acum[curr], satConnection.stateMachines[target].fsm) as (arg: string) => void;
            upAction(message);
        } catch (err) {
            Log.error(`Failed to run ${action} in ${target}`);
        }
    }

    private _mergeDefaultConfigWithCustom(machines: any[], customList: any[]): any[] {
        const updated = [];
        machines.forEach((machine) => {
            const custom = customList.find((i) => i.id === machine._id);
            if (!custom) {
                updated.push(machine);
                return;
            }
            const keys = Object.keys(custom).filter((k) => !['id', 'number'].includes(k));
            keys.forEach((k) => {
                let accessor = machine;
                const list = k.split('-');
                const last = list.pop();
                list.forEach((i) => {
                    accessor = accessor[i];
                });
                accessor[last] = custom[k];
            });
            // We only expect number to be used with API-Sync Machines
            if (!!custom.number) {
                const name = machine.config.id;
                const number = custom.number || 1;
                for (let cnt = 0; cnt < number; cnt++) {
                    const newMachine = { ...machine };
                    const newConfig = {  ...newMachine.config  };
                    const newContext = { ...newConfig.context };
                    const updatedName = `${name}_${cnt}`;
                    newContext.fsmName = updatedName;
                    newConfig.id = updatedName;
                    newConfig.context = newContext;
                    newMachine.config = newConfig;
                    updated.push(newMachine);
                }
            } else {
                updated.push(machine);
            }
        });
        return updated;
    }
